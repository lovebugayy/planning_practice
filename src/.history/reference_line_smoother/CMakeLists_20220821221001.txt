project(Reference_line_smoother)

cmake_minimum_required(VERSION 2.80)
set(CMAKE_CXX_STANDARD 17)

find_package(OsqpEigen REQUIRED)
find_package(Eigen3 REQUIRED)
find_package(Boost REQUIRED)

find_package(Python2 COMPONENTS Development Numpy)

set(PYTHON2.7_INLCUDE_DIRS "/usr/include/python2.7")


include_directories(${EIGEN3_INCLUDE_DIR})
include_directories(${OsqpEigen_INCLUDE_DIR})
include_directories(${Boost_INCLUDE_DIR})

include_directories(
    ${PYTHON2.7_INLCUDE_DIRS}
    )


include_directories(${CMAKE_SOURCE_DIR}/include)
include_directories(log)
add_library(optimization_base_lib STATIC ${CMAKE_SOURCE_DIR}/src/optimization_base.cpp
                                    ${CMAKE_SOURCE_DIR}/src/fem_reference_line_smoother.cpp)
target_link_libraries(optimization_base_lib OsqpEigen::OsqpEigen ${Boost_LIBRARIES} boost_log)

target_include_directories(${PROJECT_NAME}_main PRIVATE ${Python2_INCLUDE_DIRS} ${Python2_NumPy_INCLUDE_DIRS})
add_executable(${PROJECT_NAME}_main ${CMAKE_SOURCE_DIR}/src/main.cpp)
target_link_libraries(${PROJECT_NAME}_main optimization_base_lib  Python2::Python Python2::NumPy)
